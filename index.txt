const audio = document.getElementById("audio");

const playlist = document.getElementById("playlist");
let actual = 0;

function cargarCancion(index) {
  const cancion = canciones[index];
  audio.src = cancion.archivo;
  title.textContent = cancion.titulo;
  marcarActual(index);
  audio.play();
}

function marcarActual(index) {
  document.querySelectorAll("#playlist li").forEach((li, i) => {
    li.classList.toggle("active", i === index);
  });
}

canciones.forEach((cancion, index) => {
  const li = document.createElement("li");
  li.textContent = cancion.titulo;
  li.onclick = () => {
    actual = index;
    cargarCancion(index);
  };
  playlist.appendChild(li);
});

function siguiente() {
  actual = (actual + 1) % canciones.length;
  cargarCancion(actual);
}

function anterior() {
  actual = (actual - 1 + canciones.length) % canciones.length;
  cargarCancion(actual);
}

function togglePlay() {
  if (audio.paused) audio.play();
  else audio.pause();
}

function descargarWallpaper() {
  const ruta = 'fondopantalla.png';

(function initBannerCarousel() {
  const bannerTrack = document.getElementById('banner-track');
  const bannerSlides = Array.from(bannerTrack.children);
  const bannerNext = document.getElementById('banner-next');
  const bannerPrev = document.getElementById('banner-prev');
  const bannerIndicators = document.getElementById('banner-indicators');
  let bannerIndex = 0;
  let bannerAutoSlide;

  bannerSlides.forEach((_, i) => {
    const dot = document.createElement('div');
    dot.classList.add('dot');
    if (i === 0) dot.classList.add('active');
    dot.addEventListener('click', () => moveBannerTo(i));
    bannerIndicators.appendChild(dot);
  });
  const bannerDots = Array.from(bannerIndicators.children);

  function moveBannerTo(idx) {
    bannerIndex = (idx + bannerSlides.length) % bannerSlides.length;
    bannerTrack.style.transform = `translateX(-${bannerIndex * 100}%)`;
    bannerDots.forEach((d, i) => d.classList.toggle('active', i === bannerIndex));
  }

  bannerNext.addEventListener('click', () => moveBannerTo(bannerIndex + 1));
  bannerPrev.addEventListener('click', () => moveBannerTo(bannerIndex - 1));

  function startBannerAuto() {
    bannerAutoSlide = setInterval(() => moveBannerTo(bannerIndex + 1), 5000);
  }
  function stopBannerAuto() {
    clearInterval(bannerAutoSlide);
  }
  startBannerAuto();

  let startX = 0;
  bannerTrack.addEventListener('touchstart', (e) => {
    startX = e.touches[0].clientX;
    stopBannerAuto();
  });
  bannerTrack.addEventListener('touchend', (e) => {
    const endX = e.changedTouches[0].clientX;
    if (startX - endX > 50) moveBannerTo(bannerIndex + 1);
    else if (endX - startX > 50) moveBannerTo(bannerIndex - 1);
    startBannerAuto();
  });
})();

(function initProductCarousel() {
  const prodTrack = document.getElementById('product-track');
  const prodItems = Array.from(prodTrack.children);
  const prodNext = document.getElementById('prod-next');
  const prodPrev = document.getElementById('prod-prev');
  let prodIndex = 0;
  let prodAuto;
  let visible = computeVisible();

  function computeVisible() {
    const w = window.innerWidth;
    if (w <= 480) return 1;
    if (w <= 768) return 2;
    return 4;
  }

  function updateProdLayout() {
    visible = computeVisible();
    prodIndex = Math.min(prodIndex, Math.max(0, prodItems.length - visible));
    const shiftPercent = (prodIndex * (100 / visible));
    prodTrack.style.transform = `translateX(-${shiftPercent}%)`;
  }

  function moveProdTo(idx) {
    const maxIndex = Math.max(0, prodItems.length - visible);
    prodIndex = Math.min(maxIndex, Math.max(0, idx));
    const shiftPercent = (prodIndex * (100 / visible));
    prodTrack.style.transform = `translateX(-${shiftPercent}%)`;
  }

  prodNext.addEventListener('click', () => moveProdTo(prodIndex + 1));
  prodPrev.addEventListener('click', () => moveProdTo(prodIndex - 1));

  function startProdAuto() {
    stopProdAuto();
    prodAuto = setInterval(() => {
      const maxIndex = Math.max(0, prodItems.length - visible);
      if (prodIndex >= maxIndex) moveProdTo(0);
      else moveProdTo(prodIndex + 1);
    }, 6000);
  }

  function stopProdAuto() {
    if (prodAuto) clearInterval(prodAuto);
  }

  startProdAuto();

  let prodStartX = 0;
  prodTrack.addEventListener('touchstart', (e) => {
    prodStartX = e.touches[0].clientX;
    stopProdAuto();
  });

  prodTrack.addEventListener('touchend', (e) => {
    const prodEndX = e.changedTouches[0].clientX;
    if (prodStartX - prodEndX > 50) moveProdTo(prodIndex + 1);
    else if (prodEndX - prodStartX > 50) moveProdTo(prodIndex - 1);
    startProdAuto();
  });

  window.addEventListener('resize', updateProdLayout);
  updateProdLayout();

})();
